### 任务进度追踪

```typescript
// 实时更新任务进度
function updateMissionProgress(
  eventId: string,
  missionId: string,
  increment: number,
) {
  // 增加进度
  mission.progress += increment
  
  // 检查是否完成
  if (mission.progress >= mission.target) {
    mission.completed = true
    mission.claimed = false // 可以领取
  }
  
  // 保存到localStorage
  saveEventProgress()
}
```

### 奖励领取

```typescript
// 领取任务奖励
function claimMissionReward(eventId: string, missionId: string): number {
  if (!mission.completed || mission.claimed) return 0
  
  mission.claimed = true
  saveEventProgress()
  
  return mission.reward
}
```
